\compo{1}{6espaceexo34}{1}{On a représenté ci-contre un cube
  $ABCDEFGH$ dont les arêtes mesurent 6~cm.
  \begin{myenumerate}
    \item Dessine {\em en vraie grandeur} la face $EFAD$.
    \item Quelle est la forme géométrique particulière du triangle
      $FGH$ ? Explique pourquoi.
  \end{myenumerate}
}
%@Correction:
\begin{myenumerate}
  \setcounter{enumi}{1}
  \item Comme la face $EFGH$ est un carré alors le triangle $FGH$ est
    rectangle et isocèle en $F$.
\end{myenumerate}
