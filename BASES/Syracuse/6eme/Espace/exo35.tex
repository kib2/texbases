\compo{1}{6espaceexo35}{1}{On a représenté ci-contre un pavé droit
  $ABCDEFGH$ dont les dimensions sont indiquées sur la figure.
  \begin{myenumerate}
    \item Les droites $(GB)$ et $(AD)$ sont-elles sécantes ?
    \item Quelle est la forme géométrique particulière du triangle
      $FGH$ ? Explique pourquoi.
    \item Quelle est la forme géométrique particulière de la face
      $AFGB$ ? Explique pourquoi.
  \end{myenumerate}
}
%@Correction:
\begin{myenumerate}
  \item Elles ne sont pas sécantes.
  \item Comme la face $EFGH$ est un rectangle alors le triangle $FGH$ est
    rectangle en $F$.
  \item La face $AFGB$ est un rectangle dont tous les côtés mesurent
    4~cm. Donc $AFGB$ est un carré.
\end{myenumerate}
